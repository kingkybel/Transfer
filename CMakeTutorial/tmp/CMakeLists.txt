function(dos2unix filename)
        execute_process(COMMAND bash -c "dos2unix ${filename}"
                        OUTPUT_VARIABLE DOS2UNIX_SUCCESS)
endfunction()

dos2unix(../file1.cc)
dos2unix(../file2.cc)

endforeach()
